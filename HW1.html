manual_t_test <- function(x1, x2) {
  mean1 <- mean(x1)
  mean2 <- mean(x2)
  var1 <- var(x1)
  var2 <- var(x2)
  n1 <- length(x1)
  n2 <- length(x2)

t_stat <- (mean1 - mean2) / sqrt((var1/n1) + (var2/n2))
df <- ((var1/n1 + var2/n2)^2) / (((var1/n1)^2 / (n1 - 1)) + ((var2/n2)^2 / (n2 - 1)))
p_value <- 2 * pt(-abs(t_stat), df)
return(list(t_statistic = t_stat, p_value = p_value))
}
t_1_manual <- manual_t_test(wine_data$Total_phenols[wine_data$Class == 1],

I used the interquartile range (IQR) method to identify outliers, which is a common and effective statistical approach. First, I calculated the first quartile (Q1) and third quartile (Q3) of the salary data, then determined the IQR. To define outliers, I set the lower bound as Q1 - 1.5 * IQR and the upper bound as Q3 + 1.5 * IQR. This approach has the advantage of being based on the data’s distribution, rather than relying on fixed thresholds. The final sample size is 22590.

isPalindromic <- function(num) {
  if (!is.numeric(num) || num <= 0 || floor(num) != num) {
    stop("Input must be a positive integer.")
  }
  
  num_str <- as.character(num)
  reversed_str <- paste(rev(strsplit(num_str, "")[[1]]), collapse = "")
  is_palindrome <- (num_str == reversed_str)
  
  return(list(isPalindromic = is_palindrome, reversed = as.numeric(reversed_str)))
}

nextPalindrome <- function(num) {
  if (!is.numeric(num) || num <= 0 || floor(num) != num) {
    stop("Input must be a positive integer.")
  }
  
  next_num <- num + 1
  while (TRUE) {
    result <- isPalindromic(next_num)
    if (result$isPalindromic) {
      return(next_num)
    }
    next_num <- next_num + 1
  }
}
