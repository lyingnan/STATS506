set.seed(123)
play_dice1 <- function(n) {
  total <- 0
  for (i in 1:n) {
    roll <- sample(1:6, 1)
    if (roll %in% c(3, 5)) {
      total <- total + 2 * roll - 2
    } else {
      total <- total - 2
    }
  }
  return(total)
}

set.seed(123)
play_dice2 <- function(n) {
  rolls <- sample(1:6, n, replace = TRUE)
  wins <- sum((rolls == 3 | rolls == 5) * (2 * rolls)) 
  losses <- n * 2
  return(wins - losses)
}

# test 
print(play_dice2(3))  
[1] 6

play_dice3 <- function(n) {
  rolls <- sample(1:6, n, replace = TRUE)
  t <- table(factor(rolls, levels = 1:6))
  wins <- sum(t[c(3, 5)] * c(6, 10))
  losses <- n * 2
  return(wins - losses)
}

play_dice4 <- function(n) {

  results <- sapply(rolls, function(x) if(x %in% c(3, 5)) 2*x-2 else -2)
